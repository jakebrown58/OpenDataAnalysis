

public class SeriesFinder
{
	private List<Series> FindSeriesByName( Transactions transactions )
	{
			
		  Dictionary<string, List<Transaction>> d = 
GeneratePivotData<string, IEnumerable<Transaction>>(
		transactions,
                (x) => x.Name);

		List<Series> s = new List<Series>();

		foreach( string name in d.Keys )
		{
			Series series = SeriesFactory.BuildFrom( d[name] ));
			s.Add( series );
		}
		return s;
	}
}
